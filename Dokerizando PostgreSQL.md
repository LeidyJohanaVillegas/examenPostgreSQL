          PGADMIN_DEFAULT_PASSWORD: ${PGADMIN_DEFAULT_PASSWORD}
          PGADMIN_CONFIG_ENHANCED_COOKIE_PROTECTION: "True"
          PGADMIN_CONFIG_CONSOLE_LOG_LEVEL: "10"
